    2.如果老员工 35 岁了，也裁掉。（元素离开窗口）

int* maxSlidingWindow(int* nums, int numsSize, int k, int* returnSize) {
    *returnSize = numsSize - k + 1; // 总共有多少窗口个数，这个是可以提前确定的
    int* ans = (int*)malloc((*returnSize) * sizeof(int));
    int* q = (int*)malloc(numsSize * sizeof(int)); // 用数组模拟双端队列
    int head = 0, tail = -1; // 双端队列的队首和队尾，分别指向队首队尾元素

    // 这个单调队列应该是严格递减的——因为相同是只需要存较新的就行，老的则删除
    for(int i = 0; i < numsSize; i++){
        // 1.右边入
        while(head <= tail && nums[q[tail]] <= nums[i]) // 队列为空或者来的是严格小于队尾的元素才直接跳过，否则就是处理到新元素加入后仍保持严格递减为止或者处理到空为止-新元素入队成为唯一的元素
            tail--; // 维护q的单调性
        q[++tail] = i; // 注意保存的是下标，这样下面可以判断队首是否是有效的即是否在窗口内也即是否大于等于左端点

        // 2.左边出
        int left = i - k + 1; // 窗口左端点（因为是闭区间所以+1）
        if(q[head] < left) // 队首离开窗口，即队首的元素即下标小于左端点，即不大于等于左端点，则一直处理到满足条件(大于等于左端点)为止
            head++;

        // 3.在窗口左端点处记录答案
        if(left >= 0) // 由于队首到队尾单调递减，且队首已经在2中处理过，所以窗口最大值就是队首对应的元素
            ans[left] = nums[q[head]];
    }

    free(q);
    return ans;
}

            kinds++;
        diff[c]--; // t中字母的计数对应减-
    }

    int ans_left = -1, ans_right = INT_MAX / 2;
    int ge_cnt = 0; // 窗口内有ge_cnt种字母的出现次数>=t中相应字母的出现次数

## 83.[[Medium]56. 合并区间](https://leetcode.cn/problems/merge-intervals/description/)——“数组”&“排序”

#define MY_MAX(a, b) ((a) > (b) ? (a) : (b))
int cmp(const void* a, const void* b){ // C强制规定，必须用void*来接收任意类型数据的地址
    int* elem1 = *(int**)a, * elem2 = *(int**)b;
    return elem1[0] - elem2[0]; // 左减右意为按升序排序
}
int** merge(int** intervals, int intervalsSize, int* intervalsColSize, int* returnSize, int** returnColumnSizes) {
    qsort(intervals, intervalsSize, sizeof(int*), cmp);
    int** ans = (int**)malloc(intervalsSize * sizeof(int*)); // 分配答案数组外层空间
    *returnSize = 0;
    *returnColumnSizes = (int*)malloc(intervalsSize * sizeof(int));
    
    ans[*returnSize] = (int*)malloc(2 * sizeof(int)); // 答案数组内层空间也得分配
    ans[*returnSize][0] = intervals[0][0], ans[*returnSize][1] = intervals[0][1];
    (*returnColumnSizes)[(*returnSize)++] = 2;

    for(int i = 1; i < intervalsSize; i++){
        if(intervals[i][0] <= ans[*returnSize - 1][1]){
            // ans[*returnSize - 1][1] = intervals[i][1]; // 合并的前提下，若是新的右边界小于当前右边界不应该直接替换，即每次更新取较大的右边界
            ans[*returnSize - 1][1] = MY_MAX(ans[*returnSize - 1][1], intervals[i][1]);
            continue;
        }else{ // intervals[i][0] > ans[*returnSize - 1][1]
            ans[*returnSize] = (int*)malloc(2 * sizeof(int));
            ans[*returnSize][0] = intervals[i][0], ans[*returnSize][1] = intervals[i][1];
            (*returnColumnSizes)[(*returnSize)++] = 2; // ！！！'[]'优先级高于'*'，所以必须给解引用符号加小括号
        }
    }

    return ans;
}

## 87.[[Hard]41. 缺失的第一个正数](https://leetcode.cn/problems/first-missing-positive/description/)——“数组”&==“原地哈希”==

# ==什么是原地哈希==

**原地哈希**是一种算法技巧，指**利用数组本身作为哈希表**，通过修改数组元素来存储信息，而不使用额外的数据结构

**常见实现方式**

1. **位置交换法（值映射到索引）**
2. **符号标记法（用正负号）**

| 特征           | 说明                         |
| :------------- | :--------------------------- |
| **空间复杂度** | ==**O(1)，不使用额外数组**== |
| **修改原数组** | 通过修改数组元素来存储信息   |
| **索引映射**   | 利用数组索引和值的对应关系   |
| **信息编码**   | 将额外信息"编码"到数组元素中 |

****

int firstMissingPositive(int* nums, int numsSize) {
    for(int i = 0; i < numsSize; i++){
        // while：如果当前学生的学号在[1,n]中（为什么是1到n：因为这个数组只能装下n个，n即numsSize），但真身没有坐在正确的座位上
        while(1 <= nums[i] && nums[i] <= numsSize && nums[nums[i] - 1] != nums[i]){ // ！！！注意：这里检查的是"当前元素对应的正确位置上是否已经是当前元素"了，若是就不交换，不是才交换，！而不等价于检查当前元素是否应该在这个位置上，因为这样的话在交换的位置上元素已经等于该元素的时候会造成死循环，不仅如此，这样也不符合我们之前分析阶段时的思路逻辑：就是让学号在 1 到 n 中的学生，都坐到编号与自己学号相同的座位上（应该是学号匹配座位编号，而不是座位编号匹配学号）

因为题目要找的是第一个缺失的正数，我们只关心 1 到 n 之间的数字。

细步骤
步骤1：预处理 - 忽略非正数
	将所有非正数（≤0）改为 n+1（一个不可能在答案范围内的数）
步骤2：标记出现的数字
	遍历数组，对于每个数 num = abs(nums[i])：
	如果 1 <= num <= n，说明这个数字出现了
	将索引 num-1 位置的数标记为负数（取绝对值后再取负）
步骤3：找出第一个缺失的正数
	再次遍历数组，找到第一个正数的位置 i，答案就是 i+1
	如果全是负数，说明 1~n 都出现了，答案就是 n+1
*/
int firstMissingPositive(int* nums, int numsSize) {
    int n = numsSize;
    
    // 步骤1：将非正数改为 n+1
    for (int i = 0; i < n; i++) {
        if (nums[i] <= 0) {
            nums[i] = n + 1;
        }
    }
    
    // 步骤2：标记出现的数字
    for (int i = 0; i < n; i++) {
        int num = abs(nums[i]);  // 取绝对值
        if (1 <= num && num <= n) {
            // 标记索引 num-1 位置为负数
            if (nums[num - 1] > 0) {
                nums[num - 1] = -nums[num - 1];
            }
        }
    }
    
    // 步骤3：找第一个正数
    for (int i = 0; i < n; i++) {
        if (nums[i] > 0) {
            return i + 1;
        }
    }
    
    return n + 1;
}
