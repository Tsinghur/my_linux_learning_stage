## 126.[[Medium]22. 括号生成](https://leetcode.cn/problems/generate-parentheses/)——“回溯-序列构造型”

```c
// 回溯-序列构造型
// 本题核心：对于括号字符串的任意前缀，右括号的个数不能超过左括号的个数

// 法一：枚举当前位置填左括号还是右括号
char** generateParenthesis(int n, int* returnSize) {
    // 卡特兰数，C8 = 1430
    char** ans = (char**)malloc(1430 * sizeof(char*));
    *returnSize = 0;

    char* temp = (char*)calloc((2 * n + 1), sizeof(char)); // 字符串末尾以'\0'结尾，因为这题没有columnSize所以得设置默认的结尾字符'\0'
    void dfs(int left, int right){ // left'('的数量  right')'的数量
        if(right == n){ // right <= left <= n, 即此时right=left=n
            ans[*returnSize] = (char*)calloc((2 * n + 1), sizeof(char));
            memcpy(ans[*returnSize], temp, 2 * n * sizeof(char)); // 可以保证末尾空字符不被覆盖（虽然temp末尾也一定是空字符且不会被改变）
            (*returnSize)++;
            return;
        }

        if(left < n){ // 可以填左括号
            temp[left + right] = '(';
            dfs(left + 1, right);
        }

        if(right < left){ // 可以填右括号
            temp[left + right] = ')';
            dfs(left, right + 1);
        }
    }

    dfs(0, 0);
    return ans;
}
```
