```c
int** permute(int* nums, int numsSize, int* returnSize, int** returnColumnSizes) {
    int ansSize = 1;
    for(int i = 2; i <= numsSize; i++) // 即A~3^3
        ansSize *= i;
    int** ans = malloc(ansSize * sizeof(int*));
    *returnColumnSizes = malloc(ansSize * sizeof(int));
    *returnSize = 0;

    int* path = malloc(numsSize * sizeof(int));
    bool* onPath = calloc(numsSize, sizeof(bool)); // 初始为false即未被选-不在路径上
    
    // 枚举 path[i] 填什么数字(来自待选集合)
    void dfs(int i){
        if(i == numsSize){
            ans[*returnSize] = malloc(numsSize * sizeof(int));
            memcpy(ans[*returnSize], path, numsSize * sizeof(int));
            (*returnColumnSizes)[(*returnSize)++] = numsSize;
            return;
        }

        for(int j = 0; j < numsSize; j++)
            if(!onPath[j]){
                onPath[j] = true; // 为下层的递归调标记为已选上
                path[i] = nums[j]; // 注意不要混淆i与j
                dfs(i + 1);
                onPath[j] = false; // 恢复现场，这是为了本层的其他递归调用与上层的递归调用
            }
    }

    dfs(0);

    free(path);
    free(onPath);
    return ans;
}
```

## 120.[[Hard]51. N 皇后](https://leetcode.cn/problems/n-queens/description/)——“回溯-全排列”&“递归”

```c
// 回溯&递归
// 核心：每个皇后占据一个位置后又多了一行一列以及两条主副对角线不可用（随着往后推移限制会层层叠加），如何将这个逻辑转换成代码是个问题
char*** solveNQueens(int n, int* returnSize, int** returnColumnSizes) {
    // 1<=n<=9的范围内最多解数为352(9皇后)，由于C语言不方便动态扩容，所以最好这么做
    char*** ans = malloc(352 * sizeof(char**));
    *returnColumnSizes = malloc(352 * sizeof(int)); // 这个是表示答案数组的每个元素的长度，也就是二维数组矩阵的行数（因为ans的每个元素就是一个解即一个矩阵）
    *returnSize = 0;
    // *returnSize 告诉判题系统：一共有几个棋盘（比如 352）
    // returnColumnSizes[i] 告诉判题系统：第 i 个棋盘有几行

    int* queens = malloc(n * sizeof(int)); // 皇后放在(r, queens[r])，queens[]-储存每次解法的临时工作数组
    bool* col = calloc(n, sizeof(bool)); // 由于是遍历行，所以我们需要一个布尔数组来标记当前行中每个列是否被使用
    /*---------------------------------------------
    分别标记主对角线与副对角线是否被使用的布尔数组
        1.主对角线的特点 '\\\'
            行号 - 列号 = 常数，因为在同一条主对角线上移动行号与列号总是同时+1或者同时-1的，所以差值不变
        2.副对角线的特点 '///'
            行号 + 列号 = 常数，因为在同一条副对角线上移动行号与列号总是行+1列-1或者行-1列+1的，所以和不变
        3.为什么数组大小均是2 * n - 1
            拿主对角线举例：第一行每一个元素都对应一条，最行一行除了最后一个元素其余所有元素都分别对应一条，不就是n + n-1；或者将矩阵向右旋转45度，中间最长一行的数组元素个数就是主对角线数量，那么不就是矩阵对角线的长度吗，不就是1 + 2*(n-1)；副对角线同理
    */
    bool* diag1 = calloc(2 * n - 1, sizeof(bool));
    bool* diag2 = calloc(2 * n - 1, sizeof(bool));
    //---------------------------------------------
    
    //---------------------------------------------
    void dfs(int r){
        if(r == n){
            char** matrix = malloc(n * sizeof(char*)); // 存储n个行数组，每行表示皇后的位置即...Q...
            for(int i = 0; i < n; i++){
                 matrix[i] = malloc((n + 1) * sizeof(char)); // 每个解的每行就是一个字符串，字符串要以'\0'结尾
                 memset(matrix[i], '.', n); // 按字节填充'.'，因为'.'就是一个字节的ASCII字符
                 matrix[i][queens[i]] = 'Q'; // 每次queens数组盛满就代表一个解法答案
                 matrix[i][n] = '\0';
            } 
            ans[(*returnSize)] = matrix;
            (*returnColumnSizes)[(*returnSize)++] = n; // 每个解法棋盘有几行，不要忘记加括号，[]优先级大于*
            return; // 终止回溯递归
        }

        // 在(r, c) 放皇后（每一层递归的r就是固定的，所以只遍历列c）
        for(int c = 0; c < n; c++){
            int rc1 = r - c + n - 1; // 将r-c的最小值(n-1)映射到数组下标0——主对角线
            int rc2 = r + c; // 相加不会为负，无需映射——副对角线
            if(!col[c] && !diag1[rc1] && !diag2[rc2]){ // 判断能否放皇后，三个布尔数组的逻辑都是是否存在皇后-真/假，那么加上!就相反逻辑：真-没有，假-有
                queens[r] = c; // r行的皇后放在c列
                col[c] = diag1[rc1] = diag2[rc2] = true; // 皇后占用了c列与两条主副对角线
                dfs(r + 1);
                col[c] = diag1[rc1] = diag2[rc2] = false; // 回溯回来了，就还原现场，防止干扰当前层的其他调用与上层的调用
            }
        }
        // return;
    }
    //---------------------------------------------

    dfs(0); // 从第0行开始

    free(queens);
    free(col);
    free(diag1);
    free(diag2);
    return ans;
}
```

## 121.[[Medium-]17. 电话号码的字母组合](https://leetcode.cn/problems/letter-combinations-of-a-phone-number/description/)——“DFS”&“回溯-子集型”

```c
// 递归&回溯(没有明显体现出)
char* charSet[10] = {"", "", "abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"};

char** letterCombinations(char* digits, int* returnSize) {
    int n = strlen(digits);
    int ansSize = 1;
    for(int i = 0; i < n; i++){ // 累乘每个字符集合的长度
        int len;
        if(digits[i] == '7' || digits[i] == '9')
            len = 4;
        else
            len = 3;
        ansSize *= len;
    }
    char** ans = (char**)malloc(ansSize * sizeof(char*));
    *returnSize = 0; // 即ans的长度，初始为0
    char* tmp = (char*)calloc(n + 1, sizeof(char)); // 存储每种可能的临时工作数组，字符串结尾为'\0'，所以多申请一个字节空间，并且用calloc直接省去给末尾赋值为'\0'的操作

    
    void dfs(int i){ // 嵌套函数，可省去麻烦的传参，相当于上面的所有变量与数组都变成相对全局的了
        if(i == n){ // 工作数组存满就代表一种可能的出现
            ans[*returnSize] = (char*)malloc((n + 1) * sizeof(char));
            strcpy(ans[*returnSize], tmp); // 将工作数组复制进答案数组，不可直接存入ans中，因为还要在后面用呢
            (*returnSize)++; // 答案加一
            return; // 直接返回，每次找到一个正确答案也是回溯问题中的递归边界
        }

        int idx = digits[i] - '0'; // 将字符数字转换成数字即下标，也就是说现在要拿到所需的字符集合的下标索引
        char* set = charSet[idx]; // 取出这层递归调用所需的字符集合
        for(int j = 0; set[j]; j++){
            tmp[i] = set[j]; // 添加到辅助数组
            dfs(i + 1); // dfs(i + 1)其实就是取下一个字符集合里的一个字符，直至每个集合都取一个字符把辅助数组装满
            tmp[i] = '\0'; // 显示回溯，但此题中非必须，因为下次循环就会被覆盖
        }
    }

    dfs(0);

    free(tmp);
    return ans;
}
```
