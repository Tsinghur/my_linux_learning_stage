```c
// 形式2：自顶向下——一股迭代感
void dfs(struct TreeNode* node, int depth, int* ans){
    if(node == NULL)
        return;
    depth++; // depth为从根结点到当前层的深度，主函数调用时传入其初始值为0
    *ans = MAX(*ans, depth); // (类似迭代)每层调用都用当前深度与答案中的记录值作比较，若是大于答案值则更新，由于传参为ans的地址，相当于一个全局变量
    dfs(node->left, depth, ans);
    dfs(node->right, depth, ans);
}
int maxDepth(struct TreeNode* root) {
    int ans = 0;
    dfs(root, 0, &ans);
    return ans;
}
```

## 97.[[Easy]226. 翻转二叉树](https://leetcode.cn/problems/invert-binary-tree/description/)——“递归--自底向上/自顶向下”——简单的==树形DP==-递归式迭代

```c
// 递归

// 形式1：先递归再交换——即自底向上交换
struct TreeNode* invertTree(struct TreeNode* root) {
    if(!root) // 不能写if(!root->left)的原因：因为不一定是对称的，即左子树不存在但是右子树存在时，也得交换，即右子树置空，左子树为原来的右子树
        return NULL;
    
    struct TreeNode* tmp = root->left;
    root->left = root->right;
    root->right = tmp;
    invertTree(root->left);
    invertTree(root->right);

    return root;
}

// 形式2：先交换再递归——即自顶向下交换
struct TreeNode* invertTree(struct TreeNode* root){
    if(!root)
        return NULL;

    struct TreeNode* left = invertTree(root->left); // 翻转左子树
    struct TreeNode* right = invertTree(root->right); // 翻转右子树
    root->left = right;
    root->right = left;

    return root; // 返回值与传参相同即返回当前子树的根结点
}
```

## 100.[[Easy+]543. 二叉树的直径](https://leetcode.cn/problems/diameter-of-binary-tree/description/)——==“树形DP”==&“递归式迭代”——树形DP = 递归(遍历框架)+ DP(重叠子问题优化)

```c
// 树形DP
// ！！！二叉树的路径是简单路径，不能包含重复的边
// dfs(node) 的返回值：当前结点的左右子树高之和
// int dfs(struct TreeNode* root, int* ans){
//     if(!root)
//         return 0;
//     int left = dfs(root->left, ans);
//     int right = dfs(root->right, ans);
//     int height = fmax(left, right) + 1;
    
//     *ans = fmax(*ans, left + right);

//     return height;
// }

// int diameterOfBinaryTree(struct TreeNode* root) {
//     int ans = 0;
    
//     dfs(root, &ans);
    
//     return ans;
// }

// 标准写法
/*
    链：从 node 子树中的叶子节点到 node 的路径。把 node 子树中的最长链的长度，作为 dfs(node) 的返回值。根据这一定义，空节点的链长是 −1，叶子节点的链长是 0。
    直径：等价于由两条（或者一条）链拼成的路径。我们枚举二叉树的每个 node，假设直径在这里「拐弯」，也就是计算由左右两条从下面的叶子节点到 node 的链的节点值之和，去更新答案的最大值。

    问：可以在空节点返回 0 吗？这样 dfs(node.left) 后面无需写 + 1，更简洁一些。
    答：也可以，但需要注意的是，此时返回值的含义是「链长加一」而不是「链长」。这种写法仅仅是看上去简洁一些，细想一下，dfs 叶子的返回值居然是 1 而不是 0，未免有些荒谬。想一想，如果把题目改成带权树，也就是每条边有不同的边权，哪种写法更好呢？
*/
int dfs(struct TreeNode* root, int* ans){
    if(!root)
        return -1; // 对于叶子来说，链长就是 -1+1=0

    int l_len = dfs(root->left, ans) + 1; // 左子树最大链长+1
    int r_len = dfs(root->right, ans) + 1; // 右子树最大链长+1
    *ans = fmax(*ans, l_len + r_len); // 两条链拼成路径

    return fmax(l_len, r_len); // 以当前root为根结点的子树的最大链长
}

int diameterOfBinaryTree(struct TreeNode* root) {
    int ans = 0;
    dfs(root, &ans);
    
    return ans;
}
```
