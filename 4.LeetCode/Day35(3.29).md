// 法三：滑动窗口-非定长
int* findAnagrams(char* s, char* p, int* returnSize) {
    // 统计p的每种字母的出现次数
    int cnt[26] = {}; // 初始化为全0，等价于int cnt[26] = {0};
    int len_p = 0;
    for(; p[len_p]; len_p++)
        cnt[p[len_p] - 'a']++; 

    int* ans = (int*)malloc(strlen(s) * sizeof(int));
    *returnSize = 0;
    int left = 0;
    for(int right = 0; s[right]; right++){
        int c = s[right] - 'a';
        cnt[c]--; // 右端字母进入窗口，由于子串t与p公用一个哈希表，所以子串t的字母进入窗口就与p的相反为--，离开就++，所以cnt[c]=0时才代表窗口中字母c出现的次数与p中c出现的次数相等
        while(cnt[c] < 0){ // 窗口中字母c太多了，！！！并且一直移，就是为了正好移出一个c，这也是不定长滑动窗口的精髓，并非只移一次，而是一直移动到满足这个部分条件为止（题解中一般默认都知道，所以没有详细说明，导致会误解为只移动一次从而产生疑惑：移动一次怎么能确定移出的就是c呢），还有为什么不判断cnt[c]>0的情况，因为>0是为一直为了找到第一次子串的操作也即窗口长度不够时的操作——由外层for循环完成即一直移动右端点，也就是说右移只会最终使窗口中的某个元素增多也就是cnt[c]<0不是吗,所以才会有内层while操作，而当窗口长度正好时也就是找到所需要的子串时，这点无需细想（因为除了往窗口内添加字母即右端点右移也即cnt[c]--这个操作是每次必做的之外，其他的都是有条件的）（每次进来的多了就一定会处理到正好）（且每次添加元素都是前置判断，因为对于“这个不定长的窗口的长度正好为len_p时”，它其中的每个字母是不是都经历过判断的，且就算不满足的情况还有相应的操作，所以每个字母在这之后都是不多的，那所有的都不多且长度正好，不就是所需子串吗，因为有多的肯定有少的，有少的就肯定有多的，都不多也代表都不少，那就是正好）（补充：对于while(cnt[c]<0)这个循环，跳出时不就是cnt[c]=0，所以处理完之后不仅不多，且必然是正好；2.先来后到，也就是说每个同种元素的最后一个若是能顺利进入窗口就代表它这种字母数量就是正好，也就是说长度正好时，不可能有元素少，因为有元素少就代表有字母多，而有字母多又和开头说的正好规律冲突）
            cnt[s[left] - 'a']++; // 只能右移左端点，是为了寄希望于正好移出的也是字母c，别忘了更新真正的左端点字母的次数，因为它已经离开窗口了
            left++; // 因为这是个不定长的窗口，所以还得动态更新左端点
        }
        if(right - left + 1 == len_p)
            ans[(*returnSize)++] = left; // t左端点下标加入答案
    }
    return ans;
}
