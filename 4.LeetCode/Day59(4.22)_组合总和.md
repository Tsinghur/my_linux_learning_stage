## 125.[[Medium]39. 组合总和](https://leetcode.cn/problems/combination-sum/)——“回溯-子集型”

```c
// 回溯-子集型

// 法一：选或不选
// 此题特殊之处在于同一个数字可以无限制重复被选取，也就是说在选的逻辑下面i是不更新的，即当前数组元素不会变化，会默认一直选相同的元素直到出发边界条件
// 优化方向：剪枝优化——把 candidates 从小到大排序，如果递归中发现 left<candidates[i]，由于后面的数字只会更大，所以无法把 left 减小到 0，可以直接返回。
int** combinationSum(int* candidates, int candidatesSize, int target, int* returnSize, int** returnColumnSizes) {
    // p(n) 表示：将正整数 n 拆分成若干个正整数之和（不考虑加数的顺序），一共有多少种不同的拆分方法
    // 本题最多为p(40) = 37338，但是题目说明最多结果为150个
    int** ans = (int**)malloc(150 * sizeof(int*));
    *returnColumnSizes = (int*)malloc(150 * sizeof(int));
    *returnSize = 0;

    int* temp = (int*)malloc(20 * sizeof(int));
    int tempSize = 0;
    void dfs(int i, int left){ // left：还需要添加的剩余目标和
        if(left == 0){ // 找到一个合法组合
            ans[*returnSize] = (int*)malloc(tempSize * sizeof(int));
            memcpy(ans[*returnSize], temp, tempSize * sizeof(int));
            (*returnColumnSizes)[(*returnSize)++] = tempSize;
            return; // 别忘记提前返回，不然的话会一直走不选分支知道i == candidatesSize才结束，多走了几次答案中就多几个重复结果
        }

        if(i == candidatesSize || left < 0)
            return;

        // 不选
        dfs(i + 1, left);

        // 选
        temp[tempSize++] = candidates[i];
        dfs(i, left - candidates[i]); // 本题特点
        tempSize--; // 恢复现场
    }

    dfs(0, target);
    return ans;
}
```
