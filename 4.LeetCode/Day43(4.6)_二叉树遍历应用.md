int* rightSideView(struct TreeNode* root, int* returnSize) {
    int* ans = (int*)malloc(100 * sizeof(int));
    *returnSize = 0;

    void dfs(struct TreeNode* root, int depth){
        if(!root)
            return;
        if(depth == *returnSize) // 应该使用==判断是否要加入ans，而不是通过!=，使用!=判断会致使多余的错误元素加入ans
            ans[(*returnSize)++] = root->val;
        // depth++; // 或 在传参时更新depth
        dfs(root->right, depth + 1);
        dfs(root->left, depth + 1);
    }

    dfs(root, 0);
    return ans;
}

void flatten(struct TreeNode* root) {
    struct TreeNode* curr = root;
    while (curr) {
        if (curr->left) {
            // 找到左子树的最右节点
            struct TreeNode* pre = curr->left;
            while (pre->right)
                pre = pre->right;
            // 将右子树接到最右节点后面，注意：此时的pre并不是左子树前序遍历的最后一个结点，只是这样处理的话可以统一操作逻辑，若是它存在左子树的话，后面curr一直向右孩子迭代到它时自然会处理，即它的左子树的所有结点的顺序都应在它之后
            pre->right = curr->right;
            // 将左子树移到右边
            curr->right = curr->left;
            curr->left = NULL;
        }
        curr = curr->right;
    }
}
