## 101.[[Easy]108. 将有序数组转换为二叉搜索树](https://leetcode.cn/problems/convert-sorted-array-to-binary-search-tree/description/)——“递归”&“分治”

```c
// 递归&分治
struct TreeNode* arr_to_bst(int* nums, int l, int r){
    if(l > r)
        return NULL;
    int mid = l + (r - l) / 2;
    struct TreeNode* root = (struct TreeNode*)malloc(sizeof(struct TreeNode));
    root->val = nums[mid];
    root->left = arr_to_bst(nums, l, mid - 1);
    root->right = arr_to_bst(nums, mid + 1, r);

    return root;
}

struct TreeNode* sortedArrayToBST(int* nums, int numsSize) {
    return arr_to_bst(nums, 0, numsSize - 1);;
}
```

// 法三：返回子树的最小值与最大值-后序遍历（即法一的相反：约束传递-从下往上）
#define MY_MAX(a, b) ((a) > (b) ? (a) : (b))
#define MY_MIN(a, b) ((a) < (b) ? (a) : (b))

typedef struct{
    long long min; // 子树最小值
    long long max; // 子树最大值
}Pair;

Pair dfs(struct TreeNode* root){
    if(root == NULL)
        return (Pair){LLONG_MAX, LLONG_MIN}; // 注意这是将最小值设为MAX，最大值设为MIN，即空结点的一定为真，所以无约束
    Pair l = dfs(root->left);
    Pair r = dfs(root->right);
    long long x = root->val;
    // ！也可在递归完左子树之后立刻判断前半部分，如果发现不满足，就直接返回而不用递归右子树了
    if(x <= l.max || x >= r.min)
        return (Pair){LLONG_MIN, LLONG_MAX}; // 则之后所有这里的if判断都不满足，直到传回主函数中，主函数中也不满足
    return (Pair){MY_MIN(l.min, x), MY_MAX(r.max, x)}; // 为什么l.min与r.max还要与x比较，因为当前结点的左右子树可能为空，则就应该返回当前结点的值x作为当前子树的min/max
}

bool isValidBST(struct TreeNode* root) {
    return dfs(root).max != LLONG_MAX; // 若返回当前总树的最大最小值为LLONG_MAX/LLONG_MIN，则意为此树不是BST
}
